98 function calculateGasLoss() {

99 const V = parseFloat(document.getElementById('volume').value);

100 const tempInside = parseFloat(document.getElementById('tempInside').value);

101 const tempOutside = parseFloat(document.getElementById('tempOutside').value);

102 const openings = parseInt(document.getElementById('openings').value);

103 const tariff = parseFloat(document.getElementById('tariff').value);

104

105 if (isNaN(V) || V < 0 ||

106 isNaN(tempInside) ||

107 isNaN(tempOutside) ||

108 isNaN(openings) || openings < 0 ||

109 isNaN(tariff) || tariff < 0) {

110 alert('Пожалуйста, заполните все поля корректными значениями!');

111 return;

112 }

113

114 const deltaT = tempInside - tempOutside;

115 if (deltaT <= 0) {

116 alert('Температура внутри должна быть больше температуры снаружи!');

117 return;

118 }

119

120 const rho = 1.225; // кг/м³

121 const c = 1005; // Дж/(кг·°C)

122

123 const energyPerOpen = V * rho * c * deltaT;

124 const totalEnergy = energyPerOpen * openings;

125 const totalKWh = totalEnergy / 3600000;

126 const gasConsumed = totalKWh * 0.095;

127 const cost = gasConsumed * tariff;

128

129 const resultDiv = document.getElementById('result');

130 resultDiv.style.display = 'block';

131 resultDiv.innerHTML = `

132 <h2>Результаты расчета:</h2>

133 <p>Объем природного газа, необходимого для компенсации тепловых потерь: <strong>${gasConsumed.toFixed(2)}</strong> м³</p>

134 <p>Стоимость тепловых потерь при открывании дверей: <strong>${cost.toFixed(2)}</strong> рублей</p>

135 `;

136 }

137</script>
